export function t(
  locale: SupportedLocale,
  key: MessageKey,
  params?: Record<string, string | number>,
): string {
  const template = messages[locale]?.[key] ?? messages[DEFAULT_LOCALE][key];
 
  if (!params) return template;
 
  return template.replace(/\{\{(\w+)\}\}/g, (_, name: string) => {

}

  // キャッシュチェック
